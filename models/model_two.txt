model{
  #  Data are all caps
  #  Parameters all lower case

  #  Priors
  alpha ~ dnorm(0, 0.001)
  ndvi_eff ~ dnorm(0, 0.001)

  #  Random Effect on individual - again this is an uninformed prior
  sd_ind ~ dunif(0, 100)
  tau_ind <- 1/(sd_ind^2)

  for(i in 1:NIND){
    ind_eff[i] ~ dnorm(0, tau_ind)
  }

  # Linear Predictor
  for(i in 1:NOBS){
    logit(p[i]) <- alpha + ndvi_eff * NDVI[i] + ind_eff[IND[i]]
  }

  # Likelihood
  for(i in 1:NOBS){
    PRES[i] ~ dbern(p[i])
  }

}
